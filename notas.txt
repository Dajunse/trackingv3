const GET_WO_DETALLE = gql`
  query GetWODetalle($plano: String!) {
    workorderByPlano(plano: $plano) {
      id
      plano
      cantidad
      proyecto {
        proyecto
      }
      operaciones {
        id
        operacion
        totalAlmacen
        procesos {
          id
          conteoActual
          proceso {
            nombre
          }
        }
      }
    }
  }
`;
